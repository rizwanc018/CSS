/* 1. Name your anchor element */
.trigger-btn {
  anchor-name: --my-anchor;
}

/* 2. Position popover relative to anchor */
.popover {
  position-anchor: --my-anchor;
  position: absolute;
  
  /* Semantic positioning values */
  top: anchor(bottom);     /* Position below anchor */
  left: anchor(left);      /* Align to left edge */
  left: anchor(center);    /* Center horizontally */
  left: anchor(right);     /* Align to right edge */
}

/* 3. Dropdown menu example */
.dropdown {
  position-anchor: --dropdown-anchor;
  top: anchor(bottom);
  left: anchor(left);
  margin-top: 8px;
}
